17  package org.apache.log4j.chainsaw;
18  
19  import java.awt.Font;
20  
21  import javax.swing.UIManager;
22  import javax.swing.text.Document;
23  import javax.swing.text.JTextComponent;
24  import javax.swing.text.html.HTMLDocument;
25  
26  /**
27   * Apply system font and size (normal size + 1) rule if the JEditorPane document contains html.
28   */
29  public class JTextComponentFormatter
30  {
31      public static void applySystemFontAndSize(JTextComponent textComponent) {
32          Document document = textComponent.getDocument();
33          if (document instanceof HTMLDocument) {
34            Font font = UIManager.getFont("Label.font");
35            String bodyRule = "body { font-family: " + font.getFamily() + "; font-size: " + (font.getSize() + 1) + "pt; }";
36            ((HTMLDocument)document).getStyleSheet().addRule(bodyRule);
37          }
38      }
39  }
